核心发现 3: 质量(6.0/7)和商家信誉(5.82/7)是决策首要因素

查询能力矩阵

L1-基础: 多条件筛选 + CASE WHEN入门
SELECT respondent_id, gender, age_group, CASE monthly_income WHEN 2 THEN '3000-5000' END AS income_label FROM respondents WHERE gender = 1 AND age_group BETWEEN 2 AND 4 AND education >= 4 ORDER BY monthly_income DESC LIMIT 20;
L2-聚合: GROUP BY + HAVING初级
SELECT occupation, COUNT(*) AS cnt, ROUND(AVG(a.att_avg_score), 2) AS avg_attitude FROM respondents r JOIN attitude_scores a ON r.respondent_id = a.respondent_id GROUP BY r.occupation HAVING AVG(a.att_avg_score) > 4.0 AND COUNT(*) >= 10 ORDER BY avg_attitude DESC;
L3-连接: 五表INNER JOIN全画像中级
SELECT r.respondent_id, CASE r.gender WHEN 1 THEN '男' WHEN 2 THEN '女' END AS gender, c.awareness_level, pw.willing_to_try, a.att_avg_score, i.int_avg_score, d.f_quality FROM respondents r JOIN cognition c ON r.respondent_id = c.respondent_id JOIN purchase_willingness pw ON r.respondent_id = pw.respondent_id JOIN attitude_scores a ON r.respondent_id = a.respondent_id JOIN intention_scores i ON r.respondent_id = i.respondent_id JOIN decision_factors d ON r.respondent_id = d.respondent_id WHERE i.int_avg_score >= 4.0 ORDER BY i.int_avg_score DESC;

 

L4-高级: CTE + 窗口函数高级
WITH age_stats AS ( SELECT r.age_group, COUNT(*) AS total, SUM(CASE WHEN pw.willing_to_try=1 THEN 1 ELSE 0 END) AS willing, SUM(CASE WHEN pw.has_purchased=1 THEN 1 ELSE 0 END) AS purchased FROM respondents r JOIN purchase_willingness pw ON r.respondent_id=pw.respondent_id GROUP BY r.age_group ) SELECT age_group, total, willing, purchased, ROUND(100.0*willing/total,1) AS willingness_rate, ROUND(100.0*purchased/total,1) AS purchase_rate, ROUND(100.0*purchased/NULLIF(willing,0),1) AS conversion FROM age_stats ORDER BY age_group;
L4-高级: 相关子查询 + EXISTS高级
SELECT r.respondent_id, r.occupation, ROUND(a.att_avg_score,2) AS my_attitude, (SELECT ROUND(AVG(a2.att_avg_score),2) FROM attitude_scores a2 JOIN respondents r2 ON a2.respondent_id=r2.respondent_id WHERE r2.occupation = r.occupation) AS occ_avg FROM respondents r JOIN attitude_scores a ON r.respondent_id=a.respondent_id WHERE EXISTS ( SELECT 1 FROM purchase_willingness pw WHERE pw.respondent_id=r.respondent_id AND pw.willing_to_try=1 ) AND a.att_avg_score > (SELECT AVG(a2.att_avg_score) FROM attitude_scores a2 JOIN respondents r2 ON a2.respondent_id=r2.respondent_id WHERE r2.occupation=r.occupation) ORDER BY r.occupation;
L5-分析: RFM用户分段专家
SELECT CASE WHEN pw.has_purchased=1 AND pw.purchase_freq>=3 AND i.int_avg_score>=4.5 THEN '核心用户' WHEN pw.has_purchased=1 AND i.int_avg_score>=4.0 THEN '活跃用户' WHEN pw.willing_to_try=1 AND i.int_avg_score>=3.5 THEN '潜力用户' WHEN pw.willing_to_try=1 THEN '观望用户' ELSE '流失用户' END AS user_segment, COUNT(*) AS count, ROUND(AVG(i.int_avg_score),2) AS avg_intention FROM respondents r JOIN purchase_willingness pw ON r.respondent_id=pw.respondent_id JOIN intention_scores i ON r.respondent_id=i.respondent_id GROUP BY user_segment ORDER BY avg_intention DESC;
